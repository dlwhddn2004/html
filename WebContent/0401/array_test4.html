script 안에서 함수 밖에서 맨처음 선언되는 변수는 var키워드가 있거나 없거나 전역번수이다.
  모든 함수 내에서 사용 가능

배열1

리터럴(상수값)에 의한 배열 생성
var arr =["금","오렌지","바나나","참외","딸기","수박"];

배열2

생성자에 의한 배열 생성
var arr =new Array("사과","123","바나나","543.34","딸기","수박");

배열3

생성자에 의한 배열 생성
var arr =new Array();
arr[0] = "사과" arr[1] = 100; arr[2] = "딸기";
